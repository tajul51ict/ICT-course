area_rectangle(Length, Breadth, Area) :-
    Area is Length * Breadth.

celsius_to_fahrenheit(C, F) :-
    F is C * 9 / 5 + 32.

factorial(0, 1).

factorial(N, Result) :-
    N > 0,
    N1 is N - 1,
    factorial(N1, SubResult),
    Result is N * SubResult.


Queries:
?- area_rectangle(5, 8, Area).

?- celsius_to_fahrenheit(25, F).

?- factorial(5, F).
